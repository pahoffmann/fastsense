\SetKwRepeat{Do}{do}{while}
$T$ = Initial Transform\;
\vspace{0.2cm}
Transform point cloud with $T$\;
\vspace{0.2cm}
\Do{under max iterations and $E$ not converged}{
    \vspace{0.2cm}
    $E$ = Minimization of the sum of signed distances squared\;
    \vspace{0.2cm}
    \mbox{Transform point cloud with $T(E)$\;}
    \vspace{0.1cm}
}
\vspace{0.2cm}
\Return $T$\;
\end{algorithm}
